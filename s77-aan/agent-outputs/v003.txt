```python
# sorting.py
"""
Collection of classic sorting algorithms implemented in Python.
Includes: bubble sort, insertion sort, merge sort, quicksort.
All functions sort a list in ascending order.
"""

import random
from typing import List

def bubble_sort(arr: List[int]) -> List[int]:
    """Bubble sort – O(n²) average / worst case."""
    a = arr[:]  # work on a copy to avoid mutating the original
    n = len(a)
    for i in range(n):
        swapped = False
        for j in range(0, n - i - 1):
            if a[j] > a[j + 1]:
                a[j], a[j + 1] = a[j + 1], a[j]
                swapped = True
        if not swapped:
            break
    return a

def insertion_sort(arr: List[int]) -> List[int]:
    """Insertion sort – O(n²) average / worst, O(n) best (already sorted)."""
    a = arr[:]
    for i in range(1, len(a)):
        key = a[i]
        j = i - 1
        while j >= 0 and key < a[j]:
            a[j + 1] = a[j]
            j -= 1
        a[j + 1] = key
    return a

def merge_sort(arr: List[int]) -> List[int]:
    """Merge sort – O(n log n) always."""
    if len(arr) <= 1:
        return arr[:]
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return _merge(left, right)

def _merge(left: List[int], right: List[int]) -> List[int]:
    result = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    result.extend(left[i:])
    result.extend(right[j:])
    return result

def quicksort(arr: List[int]) -> List[int]:
    """Quicksort – O(n log n) average, O(n²) worst. In-place implementation."""
    a = arr[:]
    _quicksort_inplace(a, 0, len(a) - 1)
    return a

def _quicksort_inplace(a: List[int], low: int, high: int) -> None:
    if low < high:
        pi = _partition(a, low, high)
        _quicksort_inplace(a, low, pi - 1)
        _quicksort_inplace(a, pi + 1, high)

def _partition(a: List[int], low: int, high: int) -> int:
    # Lomuto partition scheme
    pivot = a[high]
    i = low - 1
    for j in range(low, high):
        if a[j] <= pivot:
            i += 1
            a[i], a[j] = a[j], a[i]
    a[i + 1], a[high] = a[high], a[i + 1]
    return i + 1


if __name__ == "__main__":
    # Example usage / test
    test_data = [random.randint(0, 100) for _ in range(10)]
    print("Original list:", test_data)

    print("Bubble sort:   ", bubble_sort(test_data))
    print("Insertion sort:", insertion_sort(test_data))
    print("Merge sort:    ", merge_sort(test_data))
    print("Quicksort:     ", quicksort(test_data))

    # Verify all produce a sorted copy
    expected = sorted(test_data)
    assert bubble_sort(test_data) == expected
    assert insertion_sort(test_data) == expected
    assert merge_sort(test_data) == expected
    assert quicksort(test_data) == expected
    print("\nAll algorithms returned correct sorted order.")
```
