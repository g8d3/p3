```python
# fibonacci.py

def fibonacci(n):
    """Return the nth Fibonacci number (F(0)=0, F(1)=1)."""
    if n < 0:
        raise ValueError("n must be non-negative")
    a, b = 0, 1
    for _ in range(n):
        a, b = b, a + b
    return a


if __name__ == "__main__":
    # Example usage
    for i in range(10):
        print(f"F({i}) = {fibonacci(i)}")
```
