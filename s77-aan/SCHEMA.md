# Version Registry — Core Schema

Every version is created by an agent (or human). Agents can also tag, link, and promote versions.

```sql
-- A version is a snapshot of the entire app at a point in time
CREATE TABLE versions (
    id          TEXT PRIMARY KEY,           -- "v001", "v002", "v003"... (auto-increment)
    parent_id   TEXT REFERENCES versions(id), -- what this version was forked from
    created_by  TEXT NOT NULL,              -- "human", "builder", "creator", "validator"
    message     TEXT NOT NULL,              -- what changed / why
    status      TEXT NOT NULL DEFAULT 'draft',  -- draft → active → archived | rejected
    path        TEXT NOT NULL,              -- filesystem path: versions/v001/
    created_at  TEXT NOT NULL DEFAULT (datetime('now')),
    live        INTEGER NOT NULL DEFAULT 0  -- 1 = currently live (only one)
);

-- Tags are flat labels, created by agents or humans
CREATE TABLE tags (
    id        INTEGER PRIMARY KEY AUTOINCREMENT,
    name      TEXT NOT NULL UNIQUE,         -- simple word: "ui", "bugfix", "experiment"
    created_by TEXT NOT NULL
);

-- Many-to-many: versions x tags
CREATE TABLE version_tags (
    version_id TEXT NOT NULL REFERENCES versions(id),
    tag_id     INTEGER NOT NULL REFERENCES tags(id),
    PRIMARY KEY (version_id, tag_id)
);

-- Tag-to-tag relations (for taxonomy, auto-generated by AI)
CREATE TABLE tag_relations (
    tag1_id    INTEGER NOT NULL REFERENCES tags(id),
    tag2_id    INTEGER NOT NULL REFERENCES tags(id),
    relation   TEXT NOT NULL,               -- "broader", "narrower", "related", "conflicts"
    PRIMARY KEY (tag1_id, tag2_id, relation)
);

-- Agent metadata: what each agent produced in each version
CREATE TABLE agent_work (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    version_id  TEXT NOT NULL REFERENCES versions(id),
    agent_type  TEXT NOT NULL,              -- "builder", "creator", "validator"
    input_spec  TEXT,                       -- what the agent was asked to do
    output_log  TEXT,                       -- what the agent produced / summary
    exit_status TEXT NOT NULL DEFAULT 'ok',  -- "ok", "failed", "skipped"
    started_at  TEXT,
    finished_at TEXT
);

-- One live version at a time
CREATE UNIQUE INDEX idx_version_live ON versions(live) WHERE live = 1;
```
