# File stuff
ls          # List files (like opening a folder)
cd          # Change directory (like clicking a folder)
cp          # Copy files
mv          # Move or rename files
rm          # Delete files (BE CAREFUL)
mkdir       # Make a directory
cat         # Read a file
nano        # Edit a file (beginner-friendly editor)

# System stuff
sudo        # Run as admin (with great power...)
apt         # Install apps (on Ubuntu/Debian)
top         # Task manager (but BETTER)

# Network stuff

curl        # Download stuff from the internet
ip a        # What's my IP?
ssh         # Connect to other computers (MAGIC)
